\section*{Iowa agriculture}

\toprule
Good & \multicolumn{5}{c}{Production of bushels, millions} \\
\midrule
Corn     & 0 & 1,500 & 3,000 & 4,500 & 6,000 \\
Soybeans & 2,000 & 1,750 & 1,500 & 1,000 & 0 \\
\bottomrule
\end{tabular}}
